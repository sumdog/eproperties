001    package net.jmatrix.eproperties.cli;
002    
003    /**
004     * Simple class for parsing command line arguments out of an argument 
005     * list. <p>
006     *
007     * The arguments can be switches (boolean), or Strings.  For string 
008     * arguments, either of the following 2 formats is valid:<br>
009     *
010     * -cargVal or<br>
011     * -c argVal<p>
012     * 
013     * For either of these 2 cases, a call go getStringArg("-c") will return 
014     * argVal.
015     *
016     * @author Paul Bemowski
017     */
018    public class ArgParser
019    {
020       String args[]=null;
021       
022       /**
023        * The constructor requires the arguements that are passed to the 
024        * main method.
025        */
026       public ArgParser(String a[]) {
027          args=a;
028       }
029    
030       public String getStringArg(String key){
031          return getStringArg(key, null);
032       }
033    
034       /**
035        * Returns the string corresponding to the requested key.
036        */
037       public String getStringArg(String key, String def) {
038          for (int i=0; i<args.length; i++) {
039             String arg=args[i];
040             
041             if (arg.equalsIgnoreCase(key) &&
042                 args.length >= i+2)
043                return args[i+1].trim();
044             
045             if (arg.startsWith(key))
046                return arg.substring(key.length()).trim();
047          }
048          return def;
049       }
050    
051       public int size() {return args.length;}
052       public String getLastArg() {return args[args.length-1];}
053    
054       /**
055        *
056        */
057       public Integer getIntegerArg(String key) {
058          String str=getStringArg(key);
059          if (str == null || str.trim().length() <= 0) return null;
060          
061          Integer val=null;
062          try {
063             int i=Integer.parseInt(str.trim());
064             val=new Integer(i);
065          }
066          catch (NumberFormatException ex) {
067             System.out.println ("NumberFormatException reading "+key+" from "+
068                                 "args.");
069          }
070          return val;
071       }
072    
073       /** */
074       public int getIntArg(String key, int def) {
075          Integer i=getIntegerArg(key);
076          if (i == null)
077             return def;
078          else
079             return i.intValue();
080       }
081    
082       /** */
083       public String toString() {
084          StringBuffer sb=new StringBuffer();
085          for (int i=0; i<args.length; i++) {
086             sb.append("  "+i+": '"+args[i]+"'\n");
087          }
088          return sb.toString();
089       }
090    
091       /**
092        * Returns a boolean indicating whether a command like switch was
093        * present.
094        */ 
095       public boolean getBooleanArg(String key) {
096          for (int i=0; i<args.length; i++) {
097             String arg=args[i];
098             if (arg.equalsIgnoreCase(key))
099                return true;
100          }
101          return false;
102       }
103    
104       /**
105        * Returns a boolean indicating whether a command like switch was
106        * present.
107        */ 
108       public boolean getBooleanArg(String key, boolean def) {
109          for (int i=0; i<args.length; i++) {
110             String arg=args[i];
111             if (arg.equalsIgnoreCase(key))
112                return true;
113          }
114          return def;
115       }
116    }
